93void* keyboardserver_thread(void* i)



94{



95    struct sched_param param;



96    param.sched_priority = 9;



97    pthread_setschedparam(pthread_self(), 2, &param);



98    printf("keyboardserver_thread(): Entered the Keyboard Server thread... \n");



99    



100    if((sockfd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {



101        printf("keyboardserver_thread(): ERROR : Could not create socket \n");



102        exit(1);



103    }



104    



105    printf("keyboardserver_thread(): Successfully created socket... \n");



106    



107    memset(&serv_addr, '0', sizeof(serv_addr));



108    



109    serv_addr.sin_family = AF_INET;



110    serv_addr.sin_addr.s_addr = htonl(INADDR_ANY);



111    serv_addr.sin_port = htons(12391);



112    



113    printf("keyboardserver_thread(): Bind socket... \n");



114    



115    // Fix : We could not run the Spider Flight Continuously. We need to wait for some time before you start the next flight.



116    // To fix the issue we are creating Socket with REUSEADDR.



117    if (setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int)) < 0)



118    {



119        printf("keyboardserver_thread(): ERROR: Failure in Reusing address.. with error: %d\n",errno);



120        exit(1);



121    }



122    



123    



124    if(bind(sockfd, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {



125        printf("keyboardserver_thread(): ERROR Connection to reference value server failed %d \n", errno);



126        exit(1);



127    }



128    else



129    {



130        printf("keyboardserver_thread(): Bind completed .. listening to the socket !\n");



131        



132        if (listen(sockfd, 10)<0)



133        {



134            printf("keyboardserver_thread(): Socket error (listen) %d\n",errno);



135            exit(1);



136        }



137        connfd = accept(sockfd, (struct sockaddr*)NULL, NULL);



138        setsockopt(connfd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));



139        setsockopt(connfd, SOL_SOCKET, SO_SNDBUF, &buff_size, sizeof(buff_size));



140        clientConnect = 1;



141        printf("keyboardserver_thread(): Received connection from the host! \n");



142    }



143    



144   ///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////// 



145    



146     if((sockfd_GUI = socket(AF_INET, SOCK_STREAM, 0)) < 0) {



147        printf("keyboardserver_thread(): ERROR : Could not create GUI socket\n");



148        exit(1);



149    }



150    



151    printf("keyboardserver_thread(): Successfully created GUI socket... \n");



152    



153    memset(&serv_addr, '0', sizeof(serv_addr));



154    



155    serv_addr.sin_family = AF_INET;



156    serv_addr.sin_addr.s_addr = htonl(INADDR_ANY);



157    serv_addr.sin_port = htons(24099);



158    



159    printf("keyboardserver_thread(): Bind GUI socket...\n");



160    



161    // Fix : We could not run the Spider Flight Continuously. We need to wait for some time before you start the next flight.



162    // To fix the issue we are creating Socket with REUSEADDR.



163    if (setsockopt(sockfd_GUI, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int)) < 0)



164    {



165        printf("keyboardserver_thread(): ERROR: Failure in Reusing address in GUI.. with error: %d\n",errno);



166        exit(1);



167    }



168    



169    



170    if(bind(sockfd_GUI, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {



171        printf("keyboardserver_thread(): ERROR Connection to GUI server failed %d \n", errno);



172        exit(1);



173    }



174    else



175    {



176        printf("keyboardserver_thread(): Bind completed .. listening to the GUI socket !\n");



177        



178        if (listen(sockfd_GUI, 10)<0)



179        {



180            printf("keyboardserver_thread(): GUI Socket error (listen) %d\n",errno);



181            exit(1);



182        }



183        connfd_GUI = accept(sockfd_GUI, (struct sockaddr*)NULL, NULL);



184        setsockopt(connfd_GUI, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));



185        setsockopt(connfd_GUI, SOL_SOCKET, SO_SNDBUF, &buff_size, sizeof(buff_size));



186        printf("keyboardserver_thread(): Received connection to GUI from the host! \n");



187    }



188    



189   ////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////



190    return NULL;



191}

193void create_referencevalserver_thread(void)



194{



195    int err;



196    pthread_t img_tid;



197    int policy;



198    struct sched_param param;



199    pthread_attr_t attr;



200    size_t stacksize;



201    



202    pthread_attr_init(&attr);



203    pthread_attr_getstacksize(&attr,&stacksize);



204    err = pthread_attr_setstacksize(&attr,PTHREAD_STACK_MIN);



205    printf("create_referencevalserver_thread(): stack size before reference value thread creation %d err=%d\n", stacksize,err);



206    



207    img_tid = pthread_self();



208    printf("create_referencevalserver_thread(): reference value server thread id = %d \n",(int) img_tid);



209    pthread_getschedparam(img_tid, &policy,&param);



210    printf("create_referencevalserver_thread(): reference value server  policy = %d pri=%d\n", (int)policy,param.sched_priority);



211    



212    // param1.sched_priority = param.sched_priority - 1;



213    // pthread_setschedparam(tid,policy,&param1);



214    err = pthread_create(&tid, &attr, &keyboardserver_thread, NULL);



215    if (err != 0)



216        printf("create_referencevalserver_thread(): ERROR: can't create thread :[%s]", strerror(err));



217    else



218        printf("create_referencevalserver_thread(): Reference Server Thread created successfully\n");



219    



220    pthread_attr_getstacksize(&attr,&stacksize);



221    printf("create_referencevalserver_thread(): stacksize after reference thread creation %d \n", stacksize);



222}

231void* extmode_thread(void* i)



232{



233    static int counter = 0;



234    counter++;



235    



236    if (counter == 1)



237    {



238        int policy;



239        struct sched_param param ;



240        



241        pthread_getschedparam(pthread_self(), &policy,&param);



242        printf("extmode_thread(): External mode thread policy = %d and priority=%d\n", (int)policy,param.sched_priority);



243        



244        /* initialize external mode */



245        rtParseArgsForExtMode(0, NULL);



246        rtmSetErrorStatus(MW_StringifyDefineFunction(MODEL,_M), 0);



247        MW_StringifyDefineFunction(MODEL,_initialize)();



248        /* External mode */



249        rtSetTFinalForExtMode(&rtmGetTFinal(MW_StringifyDefineFunction(MODEL,_M)));



250        rtExtModeCheckInit(1);



251



252        { 



253            boolean_T rtmStopReq = false;



254            rtExtModeWaitForStartPkt(MW_StringifyDefineFunction(MODEL,_M)->extModeInfo, 2,



255                    &rtmStopReq);



256            if (rtmStopReq) {



257                rtmSetStopRequested(MW_StringifyDefineFunction(MODEL,_M), true);



258            }



259        }



260        rtERTExtModeStartMsg();



261#ifdef PARROT_XCP_EXT_MODE



262        xcpExtModeClassicTriggerEnabled(0);



263#endif



264    }



265    while(runModel)



266    {



267        boolean_T rtmStopReq = false;



268        #ifdef MW_HAS_IMAGE_PROCESSING



269            rtExtModeOneStep(MW_StringifyDefineFunction(MODEL,_M->extModeInfo), 2, &rtmStopReq); // This is used when the Image processing is enabled along with multitasking. This will upload the step0() and step1() data.



270        #else



271            rtExtModeOneStep(MW_StringifyDefineFunction(MODEL,_M->extModeInfo), 1, &rtmStopReq); // This is used when only the control code is running. Multitasking is not enabled in this case and only the step() data is required to be uploaded.



272        #endif 



273        if (rtmStopReq)



274        {



275            printf("extmode_thread(): Thread received STOP command \n");



276            rtmSetStopRequested(MW_StringifyDefineFunction(MODEL,_M), true);



277            // Reason for the following code:



278            // If the user presses stop button before starting the flight code i.e before starting



279            // the execution step code.



280            rtExtModeShutdown(1);



281            MW_StringifyDefineFunction(MODEL,_terminate)();



282            runModel = 0;



283        }



284        



285        if (run_flag == 0)



286        {



287            rtExtModeShutdown(1);



288            MW_StringifyDefineFunction(MODEL,_terminate)();



289            runModel = 0;



290        }



291    }



292    



293    if (runModel == 0)



294    {



295        rtExtModeShutdown(1);



296        MW_StringifyDefineFunction(MODEL,_terminate)();



297    }



298    



299    background_task_completed = true;



300    return NULL;



301}

303void create_background_thread(void)



304{



305    int err, status;



306    pthread_attr_t attr;



307    size_t stacksize;



308    struct sched_param param;



309    



310    pthread_attr_init(&attr);



311    err = pthread_attr_setinheritsched(&attr,PTHREAD_EXPLICIT_SCHED);



312    printf("create_background_thread(): ext thread inherit scheduler option error=%d\n", err);



313    



314    status = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);



315    printf("create_background_thread(): value of detach status=%d\n", status);



316    



317    err = pthread_attr_setstacksize(&attr,PTHREAD_STACK_MIN);



318    printf("create_background_thread(): stacksize before thread creation err=%d\n", err);



319    



320    /* In case of Mambo , the policy of rsedu_control.c is set as 2 (SCHED_RR) and priority 32



321     * In case of Rolling Spider , the policy of rsedu_control.c is set as 0 (SCHED_OTHER) and priority 0



322     * Setting the policy and priority of External Mode thread as 0 (SCHED_OTHER) and priority 0 */



323    



324    status = pthread_attr_setschedpolicy(&attr, SCHED_OTHER);



325    printf("create_background_thread(): value of scheduler policy status=%d\n", status);



326



327    param.sched_priority = 0;



328    status = pthread_attr_setschedparam(&attr, &param);



329    printf("create_background_thread(): value of scheduler param status=%d\n", status);



330    



331    err = pthread_create(&ext_tid, &attr, &extmode_thread, NULL);



332    if (err != 0)



333        printf("create_background_thread(): can't extmode thread :[%s]", strerror(err));



334    else



335        printf("create_background_thread(): External mode Thread created successfully\n");



336    



337    pthread_attr_getstacksize(&attr,&stacksize);



338    printf("create_background_thread(): stacksize after ext thread creation %d \n", stacksize);



339    



340}
